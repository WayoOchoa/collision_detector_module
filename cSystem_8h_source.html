95 {
96 cv::Matx<T, 3, 3> R = cv::Matx<T, 3, 3>::eye();
97
98 T c1 = cayParamIn(0, 0);
99 T c2 = cayParamIn(1, 0);
100 T c3 = cayParamIn(2, 0);
101
102 T c1sqr = c1*c1;
103 T c2sqr = c2*c2;
104 T c3sqr = c3*c3;
105
106 T scale = T(1) + c1sqr + c2sqr + c3sqr;
107
108 R(0, 0) = 1 + c1sqr - c2sqr - c3sqr;
109 R(0, 1) = 2 * (c1*c2 - c3);
110 R(0, 2) = 2 * (c1*c3 + c2);
111 R(1, 0) = 2 * (c1*c2 + c3);
112 R(1, 1) = 1 - c1sqr + c2sqr - c3sqr;
113 R(1, 2) = 2 * (c2*c3 - c1);
114 R(2, 0) = 2 * (c1*c3 - c2);
115 R(2, 1) = 2 * (c2*c3 + c1);
116 R(2, 2) = 1 - c1sqr - c2sqr + c3sqr;
117
118 R = (1 / scale) * R;
119
120 return R;
121 }
