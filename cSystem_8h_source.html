106
107 R(0, 0) = 1 + c1sqr - c2sqr - c3sqr;
108 R(0, 1) = 2 * (c1*c2 - c3);
109 R(0, 2) = 2 * (c1*c3 + c2);

111 R(1, 1) = 1 - c1sqr + c2sqr - c3sqr;
112 R(1, 2) = 2 * (c2*c3 - c1);
113 R(2, 0) = 2 * (c1*c3 - c2);
114 R(2, 1) = 2 * (c2*c3 + c1);
115 R(2, 2) = 1 - c1sqr - c2sqr + c3sqr;
116
117 R = (1 / scale) * R;
